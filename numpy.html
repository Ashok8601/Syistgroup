Array Creation

🔹 np.array()

Create a NumPy array from a list, tuple, or nested lists.

import numpy as np
a = np.array([1, 2, 3])
print(a)
🔹 np.zeros()

Creates an array of given shape filled with zeros.

import numpy as np
z = np.zeros((2, 3))
print(z)
🔹 np.ones()

Creates an array filled with ones.

import numpy as np
o = np.ones((3, 3))
print(o)
🔹 np.empty()

Creates an uninitialized array (random garbage values).

import numpy as np
e = np.empty((2, 2))
print(e)
🔹 np.full()

Creates an array filled with a specified constant value.

import numpy as np
f = np.full((2, 2), 7)
print(f)
🔹 np.arange()

Returns evenly spaced values within a given interval.

import numpy as np
a = np.arange(0, 10, 2)
print(a)
🔹 np.linspace()

Returns evenly spaced numbers over a specified range.

import numpy as np
l = np.linspace(0, 1, 5)
print(l)
🔹 np.eye()

Creates a 2-D identity matrix with ones on the diagonal.

import numpy as np
i = np.eye(3)
print(i)
🔹 np.identity()

Creates an identity matrix of a given size (square matrix).

import numpy as np
i = np.identity(4)
print(i)
🔹 np.fromiter()

Creates a 1-D array from any iterable (like generator or range).

import numpy as np
arr = np.fromiter(range(5), dtype=int)
print(arr)
🔹 np.fromfunction()

Constructs an array by executing a function over each coordinate.

import numpy as np
f = np.fromfunction(lambda i, j: i + j, (3, 3), dtype=int)
print(f)
🔹 np.asarray()

Converts input to an array (avoids copy if already a NumPy array).

import numpy as np
lst = [1, 2, 3]
arr = np.asarray(lst)
print(arr)
🔹 np.copy()

Creates a deep copy of an array.

import numpy as np
a = np.array([1, 2, 3])
b = np.copy(a)
print(b)

Mathematical Functions

🔹 np.add()

Adds two arrays element-wise.

import numpy as np
a = np.array([1, 2])
b = np.array([3, 4])
print(np.add(a, b))
🔹 np.subtract()

Subtracts second array from the first, element-wise.

import numpy as np
a = np.array([5, 7])
b = np.array([2, 4])
print(np.subtract(a, b))
🔹 np.multiply()

Multiplies two arrays element-wise.

import numpy as np
a = np.array([2, 3])
b = np.array([4, 5])
print(np.multiply(a, b))
🔹 np.divide()

Divides first array by second, element-wise.

import numpy as np
a = np.array([10, 20])
b = np.array([2, 5])
print(np.divide(a, b))
🔹 np.power()

Raises elements of first array to powers from second array.

import numpy as np
a = np.array([2, 3])
b = np.array([3, 2])
print(np.power(a, b))
🔹 np.sqrt()

Computes the square root of each element.

import numpy as np
arr = np.array([1, 4, 9])
print(np.sqrt(arr))
🔹 np.exp()

Calculates the exponential of all elements (e^x).

import numpy as np
arr = np.array([1, 2, 3])
print(np.exp(arr))
🔹 np.log()

Computes the natural logarithm (ln) of all elements.

import numpy as np
arr = np.array([1, np.e, np.e**2])
print(np.log(arr))
🔹 np.log10()

Computes the base-10 logarithm of all elements.

import numpy as np
arr = np.array([1, 10, 100])
print(np.log10(arr))
🔹 np.log2()

Computes the base-2 logarithm of all elements.

import numpy as np
arr = np.array([1, 2, 4, 8])
print(np.log2(arr))
🔹 np.sin()

Computes the sine of each element (in radians).

import numpy as np
angles = np.array([0, np.pi/2, np.pi])
print(np.sin(angles))
🔹 np.cos()

Computes the cosine of each element (in radians).

import numpy as np
angles = np.array([0, np.pi/2, np.pi])
print(np.cos(angles))
🔹 np.tan()

Computes the tangent of each element (in radians).

import numpy as np
angles = np.array([0, np.pi/4])
print(np.tan(angles))
🔹 np.abs()

Computes the absolute value element-wise.

import numpy as np
arr = np.array([-1, -3, 4])
print(np.abs(arr))
🔹 np.ceil()

Rounds each element up to the nearest integer.

import numpy as np
arr = np.array([1.2, 2.8])
print(np.ceil(arr))
🔹 np.floor()

Rounds each element down to the nearest integer.

import numpy as np
arr = np.array([1.9, 2.1])
print(np.floor(arr))
🔹 np.round()

Rounds each element to the nearest integer.

import numpy as np
arr = np.array([1.4, 2.5, 3.6])
print(np.round(arr))
🔹 np.clip()

Clips (limits) values to a specified range.

import numpy as np
arr = np.array([1, 5, 10, 15])
print(np.clip(arr, 3, 10))
🔹 np.mod()

Computes the element-wise remainder of division.

import numpy as np
a = np.array([5, 9])
b = np.array([2, 4])
print(np.mod(a, b))
🔹 np.remainder()

Same as np.mod() — returns remainder of division.

import numpy as np
print(np.remainder([4, 7], [2, 3]))

Random Module

🔹 np.random.rand()

Generates random floats in [0, 1) with given shape.

import numpy as np
print(np.random.rand(2, 2))
🔹 np.random.randint()

Generates random integers in a given range and shape.

import numpy as np
print(np.random.randint(1, 10, (3, 3)))
🔹 np.random.randn()

Returns samples from the standard normal distribution.

import numpy as np
print(np.random.randn(2, 3))
🔹 np.random.choice()

Selects random elements from a given array.

import numpy as np
arr = np.array([10, 20, 30, 40])
print(np.random.choice(arr, 2))
🔹 np.random.shuffle()

Shuffles the contents of an array in-place.

import numpy as np
arr = np.array([1, 2, 3, 4])
np.random.shuffle(arr)
print(arr)
🔹 np.random.permutation()

Returns a permuted (shuffled) version of the array.

import numpy as np
arr = np.array([1, 2, 3, 4])
print(np.random.permutation(arr))
🔹 np.random.seed()

Sets the seed for random number generation (for reproducibility).

import numpy as np
np.random.seed(42)
print(np.random.rand(2))
🔹 np.random.uniform()

Draws samples from a uniform distribution.

import numpy as np
print(np.random.uniform(1, 5, size=(2, 3)))
🔹 np.random.normal()

Draws samples from a normal (Gaussian) distribution.

import numpy as np
print(np.random.normal(loc=0.0, scale=1.0, size=(2, 3)))

Statistics

🔹 np.mean()

Computes the arithmetic mean (average) of array elements.

import numpy as np
arr = np.array([1, 2, 3, 4])
print(np.mean(arr))
🔹 np.median()

Computes the median (middle value) of array elements.

import numpy as np
arr = np.array([1, 3, 2, 4])
print(np.median(arr))
🔹 np.std()

Computes the standard deviation, a measure of spread in the data.

import numpy as np
arr = np.array([1, 2, 3, 4])
print(np.std(arr))
🔹 np.var()

Computes the variance of array elements.

import numpy as np
arr = np.array([1, 2, 3, 4])
print(np.var(arr))
🔹 np.min()

Returns the minimum value in the array.

import numpy as np
arr = np.array([5, 2, 9, 1])
print(np.min(arr))
🔹 np.max()

Returns the maximum value in the array.

import numpy as np
arr = np.array([5, 2, 9, 1])
print(np.max(arr))
🔹 np.argmax()

Returns the index of the maximum value.

import numpy as np
arr = np.array([5, 2, 9, 1])
print(np.argmax(arr))
🔹 np.argmin()

Returns the index of the minimum value.

import numpy as np
arr = np.array([5, 2, 9, 1])
print(np.argmin(arr))
🔹 np.percentile()

Computes the q-th percentile of the data.

import numpy as np
arr = np.array([1, 2, 3, 4, 5])
print(np.percentile(arr, 50))  # Median
🔹 np.quantile()

Computes the quantiles of the data.

import numpy as np
arr = np.array([1, 2, 3, 4, 5])
print(np.quantile(arr, 0.5))  # Median
🔹 np.average()

Computes the weighted average of array elements.

import numpy as np
arr = np.array([1, 2, 3])
weights = np.array([0.2, 0.3, 0.5])
print(np.average(arr, weights=weights))
🔹 np.corrcoef()

Computes the Pearson correlation coefficient matrix.

import numpy as np
x = np.array([1, 2, 3])
y = np.array([2, 4, 6])
print(np.corrcoef(x, y))
🔹 np.cov()

Computes the covariance matrix.

import numpy as np
x = np.array([1, 2, 3])
y = np.array([2, 4, 6])
print(np.cov(x, y))
🔹 np.histogram()

Computes the histogram of the data.

import numpy as np
data = np.random.randn(1000)
hist, bins = np.histogram(data, bins=30)
print(hist)
🔹 np.bincount()

Counts number of occurrences of each value in an array of non-negative ints.

import numpy as np
arr = np.array([0,1,1,3,2,1,7])
print(np.bincount(arr))
🔹 np.ptp()

Returns the range (max - min) of values along an axis.

import numpy as np
arr = np.array([1, 3, 7, 2])
print(np.ptp(arr))
